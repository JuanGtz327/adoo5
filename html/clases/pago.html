package itzamara;

    public class pago {
        private float monto;
        private String fecha;
        private String tipo;
    
        public float getMonto() {
            return monto;
        }
    
        public void setMonto(float monto) {
            this.monto = monto;
        }
    
        public String getFecha() {
            return fecha;
        }
    
        public void setFecha(String fecha) {
            this.fecha = fecha;
        }
    
        public String getTipo() {
            return tipo;
        }
    
        public void setTipo(String tipo) {
            this.tipo = tipo;
        }
    }
